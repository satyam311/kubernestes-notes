deploy.yaml 

```
apiVersion: apps/v1
kind: Deployment
metadata:
  name: php-apache
spec:
  selector:
    matchLabels:
      run: php-apache
  template:
    metadata:
      labels:
        run: php-apache
    spec:
      containers:
      - name: php-apache
        image: registry.k8s.io/hpa-example
        ports:
        - containerPort: 80
        resources:
          limits:
            cpu: 500m
          requests:
            cpu: 200m
---
apiVersion: v1
kind: Service
metadata:
  name: php-apache
  labels:
    run: php-apache
spec:
  ports:
  - port: 80
  selector:
    run: php-apache

```

hap.yaml : 

```
apiVersion: autoscaling/v2
kind: HorizontalPodAutoscaler
metadata:
  name: php-apache-hpa
spec:
  # 1. Target Resource (The Deployment to scale)
  scaleTargetRef:
    apiVersion: apps/v1
    kind: Deployment
    name: php-apache  # Matches the Deployment name from your command
  
  # 2. Min/Max Replicas
  minReplicas: 1      # Matches --min=1
  maxReplicas: 10     # Matches --max=10
  
  # 3. Scaling Metrics (The rule for when to scale)
  metrics:
  - type: Resource
    resource:
      name: cpu
      target:
        type: Utilization
        averageUtilization: 50 # Matches --cpu-percent=50

```
or 

```kubectl autoscale deployment php-apache --cpu-percent=50 --min=1 --max=10```
